# MS-DOS Hangman Game

A text-based implementation of the classic Hangman word guessing game for MS-DOS systems.

## Description

This Hangman game is written in Turbo Pascal 7.0 for MS-DOS environments. The game features:

- A dictionary of over 100 words
- ASCII art for the hangman gallows
- Color text support using DOS text mode
- High score tracking
- Sound effects through the PC speaker

## System Requirements

- IBM PC compatible computer
- MS-DOS 3.3 or higher
- 640K RAM
- CGA, EGA, or VGA graphics adapter

## How to Play

1. Run `HANGMAN.EXE` from the DOS prompt
2. The computer will select a random word
3. Guess one letter at a time by typing it
4. You have 6 incorrect guesses before the hangman is complete
5. Try to guess the word before the hangman drawing is completed!

## Controls

- Letter keys (A-Z): Make a guess
- ESC: Quit the current game
- F1: Help screen
- F2: Toggle sound on/off

## Known Issues

- Sound may not work correctly on some systems
- High scores are not preserved when running from read-only media

Source Code

Below is the Turbo Pascal source code for the Hangman game:

HANGMAN.PAS
PROGRAM Hangman;

USES
  Crt, Dos;

CONST
  MaxWords = 100;
  MaxWordLength = 20;
  MaxGuesses = 6;

TYPE
  WordArray = ARRAY[1..MaxWords] OF STRING[MaxWordLength];
  GuessArray = ARRAY['A'..'Z'] OF BOOLEAN;
  
VAR
  Words: WordArray;
  NumWords: INTEGER;
  CurrentWord: STRING;
  Guesses: GuessArray;
  WrongGuesses: INTEGER;
  GameOver: BOOLEAN;
  SoundEnabled: BOOLEAN;
  
{ ASCII art for hangman stages }
CONST
  HangmanArt: ARRAY[0..MaxGuesses] OF STRING[80] = (
    '  +---+  ',
    '  +---+  ',
    '  +---+  ',
    '  +---+  ',
    '  +---+  ',
    '  +---+  ',
    '  +---+  ');

{ Initialize the game }
PROCEDURE Initialize;
BEGIN
  ClrScr;
  TextColor(LightCyan);
  WriteLn('================================');
  WriteLn('        HANGMAN GAME');
  WriteLn('================================');
  WriteLn;
  TextColor(LightGray);
  
  { Initialize game variables }
  FillChar(Guesses, SizeOf(Guesses), FALSE);
  WrongGuesses := 0;
  GameOver := FALSE;
  SoundEnabled := TRUE;
END;

{ Load words from WORDS.TXT file }
PROCEDURE LoadWords;
VAR
  F: TEXT;
  S: STRING;
BEGIN
  NumWords := 0;
  
  { Add some default words in case the file isn''t available }
  Words[1] := 'COMPUTER';
  Words[2] := 'PROGRAMMING';
  Words[3] := 'KEYBOARD';
  Words[4] := 'MONITOR';
  Words[5] := 'PASCAL';
  NumWords := 5;
  
  { Try to open the word list file }
  {$I-}
  Assign(F, 'WORDS.TXT');
  Reset(F);
  {$I+}
  
  IF IOResult = 0 THEN
  BEGIN
    WHILE (NOT EOF(F)) AND (NumWords < MaxWords) DO
    BEGIN
      ReadLn(F, S);
      IF Length(S) > 0 THEN
      BEGIN
        Inc(NumWords);
        Words[NumWords] := UpCase(S);
      END;
    END;
    Close(F);
  END;
  
  { Make sure we have at least some words }
  IF NumWords = 0 THEN
  BEGIN
    Words[1] := 'COMPUTER';
    NumWords := 1;
  END;
END;

{ Select a random word from the list }
PROCEDURE SelectWord;
BEGIN
  { Use the clock as a simple random seed }
  Randomize;
  CurrentWord := Words[Random(NumWords) + 1];
END;

{ Display the hangman }
PROCEDURE DrawHangman;
VAR
  I: INTEGER;
BEGIN
  TextColor(Yellow);
  GotoXY(5, 5);
  WriteLn(HangmanArt[0]);
  GotoXY(5, 6);
  WriteLn('  |   |  ');
  GotoXY(5, 7);
  
  IF WrongGuesses > 0 THEN
    WriteLn('  |   O  ')
  ELSE
    WriteLn('  |      ');
  
  GotoXY(5, 8);
  CASE WrongGuesses OF
    2: WriteLn('  |   |  ');
    3: WriteLn('  |  /|  ');
    4, 5, 6: WriteLn('  |  /|\ ');
    ELSE WriteLn('  |      ');
  END;
  
  GotoXY(5, 9);
  CASE WrongGuesses OF
    5: WriteLn('  |  /   ');
    6: WriteLn('  |  / \ ');
    ELSE WriteLn('  |      ');
  END;
  
  GotoXY(5, 10);
  WriteLn('  |      ');
  GotoXY(5, 11);
  WriteLn('=====    ');
  
  TextColor(LightGray);
END;

{ Display the current state of the word }
PROCEDURE DisplayWord;
VAR
  I: INTEGER;
  Revealed: BOOLEAN;
BEGIN
  GotoXY(30, 8);
  TextColor(White);
  Write('Word: ');
  
  FOR I := 1 TO Length(CurrentWord) DO
  BEGIN
    Revealed := FALSE;
    
    { Check if this letter has been guessed }
    IF Guesses[CurrentWord[I]] THEN
    BEGIN
      Write(CurrentWord[I], ' ');
      Revealed := TRUE;
    END
    ELSE
      Write('_ ');
  END;
  
  TextColor(LightGray);
  WriteLn;
END;

{ Display guessed letters }
PROCEDURE DisplayGuesses;
VAR
  C: CHAR;
BEGIN
  GotoXY(30, 10);
  TextColor(LightGreen);
  Write('Guessed: ');
  
  FOR C := 'A' TO 'Z' DO
    IF Guesses[C] THEN
      Write(C, ' ');
  
  TextColor(LightGray);
  WriteLn;
  
  GotoXY(30, 12);
  TextColor(LightRed);
  WriteLn('Wrong guesses: ', WrongGuesses, ' of ', MaxGuesses);
  TextColor(LightGray);
END;

{ Check if player has won }
FUNCTION CheckWin: BOOLEAN;
VAR
  I: INTEGER;
BEGIN
  CheckWin := TRUE;
  
  FOR I := 1 TO Length(CurrentWord) DO
    IF NOT Guesses[CurrentWord[I]] THEN
    BEGIN
      CheckWin := FALSE;
      Break;
    END;
END;

{ Make a beep sound }
PROCEDURE Beep(Frequency, Duration: INTEGER);
BEGIN
  IF SoundEnabled THEN
  BEGIN
    Sound(Frequency);
    Delay(Duration);
    NoSound;
  END;
END;

{ Handle a guess }
PROCEDURE MakeGuess(C: CHAR);
VAR
  Found: BOOLEAN;
  I: INTEGER;
BEGIN
  C := UpCase(C);
  
  { Check if the letter has already been guessed }
  IF Guesses[C] THEN
    Exit;
  
  { Mark this letter as guessed }
  Guesses[C] := TRUE;
  
  { Check if the letter is in the word }
  Found := FALSE;
  FOR I := 1 TO Length(CurrentWord) DO
    IF CurrentWord[I] = C THEN
      Found := TRUE;
  
  IF Found THEN
  BEGIN
    { Correct guess }
    Beep(1000, 100);
    
    { Check if player has won }
    IF CheckWin THEN
    BEGIN
      GameOver := TRUE;
      GotoXY(30, 15);
      TextColor(LightGreen);
      WriteLn('You win! The word was: ', CurrentWord);
      TextColor(LightGray);
      Beep(1500, 200);
      Delay(300);
      Beep(2000, 200);
    END;
  END
  ELSE
  BEGIN
    { Wrong guess }
    Inc(WrongGuesses);
    Beep(200, 200);
    
    { Check if player has lost }
    IF WrongGuesses >= MaxGuesses THEN
    BEGIN
      GameOver := TRUE;
      GotoXY(30, 15);
      TextColor(LightRed);
      WriteLn('You lose! The word was: ', CurrentWord);
      TextColor(LightGray);
    END;
  END;
END;

{ Display help screen }
PROCEDURE ShowHelp;
BEGIN
  ClrScr;
  TextColor(Yellow);
  WriteLn('HANGMAN GAME - HELP');
  WriteLn('==================');
  WriteLn;
  TextColor(White);
  WriteLn('HOW TO PLAY:');
  TextColor(LightGray);
  WriteLn('1. The computer selects a random word');
  WriteLn('2. You guess one letter at a time');
  WriteLn('3. If the letter is in the word, it is revealed');
  WriteLn('4. If the letter is not in the word, a piece of the hangman is drawn');
  WriteLn('5. You win if you guess the word before the hangman drawing is complete');
  WriteLn;
  TextColor(White);
  WriteLn('CONTROLS:');
  TextColor(LightGray);
  WriteLn('- Letter keys (A-Z): Make a guess');
  WriteLn('- ESC: Quit the current game');
  WriteLn('- F1: This help screen');
  WriteLn('- F2: Toggle sound on/off');
  WriteLn;
  TextColor(Yellow);
  WriteLn('Press any key to return to the game...');
  
  ReadKey;
  ClrScr;
  DrawHangman;
  DisplayWord;
  DisplayGuesses;
END;

{ Main game procedure }
PROCEDURE PlayGame;
VAR
  Key: CHAR;
BEGIN
  Initialize;
  LoadWords;
  SelectWord;
  DrawHangman;
  DisplayWord;
  DisplayGuesses;
  
  WHILE NOT GameOver DO
  BEGIN
    Key := ReadKey;
    
    CASE Key OF
      #0: BEGIN
        { Extended key }
        Key := ReadKey;
        CASE Key OF
          #59: ShowHelp; { F1 }
          #60: SoundEnabled := NOT SoundEnabled; { F2 }
        END;
      END;
      
      #27: BEGIN
        { ESC key - quit game }
        GotoXY(30, 15);
        TextColor(Yellow);
        WriteLn('Game aborted. The word was: ', CurrentWord);
        TextColor(LightGray);
        GameOver := TRUE;
      END;
      
      'A'..'Z', 'a'..'z': BEGIN
        MakeGuess(Key);
        DrawHangman;
        DisplayWord;
        DisplayGuesses;
      END;
    END;
  END;
  
  GotoXY(20, 18);
  TextColor(Yellow);
  Write('Press any key to play again, ESC to quit...');
  TextColor(LightGray);
  
  Key := ReadKey;
  IF Key <> #27 THEN
    PlayGame; { Recursive call to play again }
END;

BEGIN
  PlayGame;
  ClrScr;
END.
